REPLIT INSTRUCTIONS: INSTANT REFERRAL COMMISSION PAYOUTS
My Car Concierge Platform

## OVERVIEW

Configure instant payout system for Member Founders and Pilot Provider Founders
who earn 50% commission when their referrals purchase bid credit packs. Payments
are triggered automatically via Stripe webhook when a purchase completes.

## Step 1.1: Add Referral Tracking Tables

Execute the following SQL in your Supabase SQL Editor:

– Track referral relationships
CREATE TABLE referrals (
id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
referrer_id UUID NOT NULL REFERENCES users(id),
referred_id UUID NOT NULL REFERENCES users(id),
referrer_type TEXT NOT NULL,
referred_type TEXT NOT NULL,
referral_code TEXT,
status TEXT DEFAULT ‘active’,
created_at TIMESTAMPTZ DEFAULT NOW(),

UNIQUE(referrer_id, referred_id)
);

CREATE INDEX idx_referrals_referrer ON referrals(referrer_id);
CREATE INDEX idx_referrals_referred ON referrals(referred_id);
CREATE INDEX idx_referrals_code ON referrals(referral_code);

– Track all commission payments
CREATE TABLE commission_payments (
id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
referrer_id UUID NOT NULL REFERENCES users(id),
referred_id UUID NOT NULL REFERENCES users(id),
purchase_id UUID NOT NULL,
stripe_payment_intent_id TEXT,

purchase_amount DECIMAL(10,2) NOT NULL,
commission_rate DECIMAL(5,4) NOT NULL,
commission_amount DECIMAL(10,2) NOT NULL,

stripe_transfer_id TEXT,
stripe_payout_id TEXT,
payout_method TEXT,
payout_fee DECIMAL(10,2) DEFAULT 0,
net_payout DECIMAL(10,2),

status TEXT DEFAULT ‘pending’,
paid_at TIMESTAMPTZ,
failed_reason TEXT,

created_at TIMESTAMPTZ DEFAULT NOW(),
updated_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_commission_payments_referrer ON commission_payments(referrer_id);
CREATE INDEX idx_commission_payments_status ON commission_payments(status);
CREATE INDEX idx_commission_payments_purchase ON commission_payments(purchase_id);

## Step 4.1: Create Commission Calculator

File: server/stripe/commissions.js

import Stripe from ‘stripe’;
const stripe = new Stripe(process.env.STRIPE_SECRET_KEY);

const COMMISSION_RATE = 0.50;

export function calculateCommission(purchaseAmount) {
return parseFloat((purchaseAmount * COMMISSION_RATE).toFixed(2));
}

export function calculateInstantPayoutFee(amount) {
const fee = amount * 0.01;
return parseFloat(Math.max(0.50, Math.min(10, fee)).toFixed(2));
}

export async function processInstantCommission(commissionPaymentId) {
try {
const { data: commission, error: commError } = await supabase
.from(‘commission_payments’)
.select(`*, referrer:referrer_id(stripe_connect_account_id, payout_preference, is_member_founder, is_provider_founder)`)
.eq(‘id’, commissionPaymentId)
.single();

```
if (commError || !commission) {
  throw new Error('Commission payment not found');
}

const referrer = commission.referrer;

if (!referrer.stripe_connect_account_id) {
  throw new Error('Referrer has no Stripe Connect account');
}

const accountStatus = await stripe.accounts.retrieve(referrer.stripe_connect_account_id);
if (!accountStatus.payouts_enabled) {
  throw new Error('Referrer payouts not enabled');
}

const useInstantPayout = referrer.payout_preference === 'instant';
const payoutFee = useInstantPayout ? calculateInstantPayoutFee(commission.commission_amount) : 0;
const netPayout = commission.commission_amount - payoutFee;

const transfer = await stripe.transfers.create({
  amount: Math.round(netPayout * 100),
  currency: 'usd',
  destination: referrer.stripe_connect_account_id,
  metadata: {
    commission_payment_id: commissionPaymentId,
    referrer_id: commission.referrer_id,
    purchase_id: commission.purchase_id,
    founder_type: referrer.is_member_founder ? 'member_founder' : 'provider_founder',
  },
});

let payout = null;
if (useInstantPayout) {
  payout = await stripe.payouts.create(
    {
      amount: Math.round(netPayout * 100),
      currency: 'usd',
      method: 'instant',
    },
    {
      stripeAccount: referrer.stripe_connect_account_id,
    }
  );
} else {
  payout = await stripe.payouts.create(
    {
      amount: Math.round(netPayout * 100),
      currency: 'usd',
      method: 'standard',
    },
    {
      stripeAccount: referrer.stripe_connect_account_id,
    }
  );
}

await supabase
  .from('commission_payments')
  .update({
    status: 'paid',
    stripe_transfer_id: transfer.id,
    stripe_payout_id: payout.id,
    payout_method: useInstantPayout ? 'instant' : 'standard',
    payout_fee: payoutFee,
    net_payout: netPayout,
    paid_at: new Date().toISOString(),
    updated_at: new Date().toISOString(),
  })
  .eq('id', commissionPaymentId);

await supabase
  .from('users')
  .update({
    total_commissions_paid: supabase.raw(`total_commissions_paid + ${netPayout}`),
  })
  .eq('id', commission.referrer_id);

console.log(`✅ Commission paid: $${netPayout} to ${referrer.stripe_connect_account_id} (${useInstantPayout ? 'instant' : 'standard'})`);

return {
  success: true,
  transferId: transfer.id,
  payoutId: payout.id,
  amount: netPayout,
  method: useInstantPayout ? 'instant' : 'standard',
};
```

} catch (error) {
console.error(‘❌ Commission payout error:’, error);

```
await supabase
  .from('commission_payments')
  .update({
    status: 'failed',
    failed_reason: error.message,
    updated_at: new Date().toISOString(),
  })
  .eq('id', commissionPaymentId);

throw error;
```

}
}

## Step 5.1: Create Webhook Endpoint

File: server/webhooks/stripe.js

import Stripe from ‘stripe’;
import { calculateCommission, processInstantCommission } from ‘../stripe/commissions.js’;

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY);
const webhookSecret = process.env.STRIPE_WEBHOOK_SECRET;

export async function POST(req, res) {
const sig = req.headers[‘stripe-signature’];
let event;

try {
event = stripe.webhooks.constructEvent(req.body, sig, webhookSecret);
} catch (err) {
console.error(‘⚠️ Webhook signature verification failed:’, err.message);
return res.status(400).send(`Webhook Error: ${err.message}`);
}

switch (event.type) {
case ‘payment_intent.succeeded’:
await handlePaymentIntentSucceeded(event.data.object);
break;

```
case 'transfer.created':
  console.log('Transfer created:', event.data.object.id);
  break;

case 'payout.paid':
  console.log('Payout completed:', event.data.object.id);
  break;

case 'payout.failed':
  await handlePayoutFailed(event.data.object);
  break;

default:
  console.log(`Unhandled event type ${event.type}`);
```

}

res.json({ received: true });
}

async function handlePaymentIntentSucceeded(paymentIntent) {
try {
console.log(‘💳 Payment succeeded:’, paymentIntent.id);

```
const { data: purchase, error: purchaseError } = await supabase
  .from('bid_credit_purchases')
  .select('*')
  .eq('stripe_payment_intent_id', paymentIntent.id)
  .single();

if (purchaseError || !purchase) {
  console.log('No bid credit purchase found for payment:', paymentIntent.id);
  return;
}

await supabase
  .from('bid_credit_purchases')
  .update({ status: 'completed' })
  .eq('id', purchase.id);

if (!purchase.referred_by) {
  console.log('Purchase not referred, no commission to pay');
  return;
}

console.log(`📊 Referred purchase by user ${purchase.referred_by}`);

const commissionAmount = calculateCommission(purchase.amount_paid);

const { data: commissionPayment, error: commError } = await supabase
  .from('commission_payments')
  .insert({
    referrer_id: purchase.referred_by,
    referred_id: purchase.user_id,
    purchase_id: purchase.id,
    stripe_payment_intent_id: paymentIntent.id,
    purchase_amount: purchase.amount_paid,
    commission_rate: 0.50,
    commission_amount: commissionAmount,
    status: 'pending',
  })
  .select()
  .single();

if (commError) {
  console.error('Error creating commission payment:', commError);
  return;
}

await supabase
  .from('users')
  .update({
    total_commissions_earned: supabase.raw(`total_commissions_earned + ${commissionAmount}`),
  })
  .eq('id', purchase.referred_by);

console.log(`💰 Commission created: $${commissionAmount} for user ${purchase.referred_by}`);

await processInstantCommission(commissionPayment.id);
```

} catch (error) {
console.error(‘Error handling payment success:’, error);
}
}

async function handlePayoutFailed(payout) {
try {
console.error(‘❌ Payout failed:’, payout.id, payout.failure_message);

```
const { data: commission } = await supabase
  .from('commission_payments')
  .select('*')
  .eq('stripe_payout_id', payout.id)
  .single();

if (commission) {
  await supabase
    .from('commission_payments')
    .update({
      status: 'failed',
      failed_reason: payout.failure_message,
      updated_at: new Date().toISOString(),
    })
    .eq('id', commission.id);
}
```

} catch (error) {
console.error(‘Error handling payout failure:’, error);
}
}
